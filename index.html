// ── Revenue on MTD rep cards ────────────────────────────────────────── window._mtdRevReps = null; // holds current-month OR historical rep revenue function addRevToCards(){ var reps = window._mtdRevReps || (revData && revData.reps) || []; if(!reps.length) return; document.querySelectorAll('.rep-card').forEach(function(card){ var old=card.querySelector('.rep-rev-row'); if(old) old.parentNode.removeChild(old); var nameEl=card.querySelector('.rep-name-btn'); if(!nameEl) return; var repName=Array.from(nameEl.childNodes) .filter(function(n){return n.nodeType===3;}) .map(function(n){return n.textContent.trim();}) .join('').replace(/\s+/g,' ').trim(); var rev=reps.find(function(r){return r.rep===repName;}); if(!rev) return; // Support both revData field names and revenue-history field names var cs = rev.contract_mtd_amount !== undefined ? rev.contract_mtd_amount : (rev.contract_amount||0); var ap = rev.approved_mtd_amount !== undefined ? rev.approved_mtd_amount : (rev.approved_amount||0); var row=document.createElement('div'); row.className='rep-rev-row'; row.innerHTML='CS'+fmtD(cs)+'' +'Appr'+fmtD(ap)+''; card.appendChild(row); }); } // Wrap renderMTD so revenue re-renders after every CA update var _omtd=renderMTD; renderMTD=function(){ _omtd.apply(this,arguments); setTimeout(addRevToCards,50); }; // Wrap changeMTDMonth to update rep cards when historical month is picked var _oChgMo=changeMTDMonth; changeMTDMonth=function(val){ window._mtdRevReps=null; // reset _oChgMo.apply(this,arguments); // Intercept the revenue-history fetch result for rep cards var now=new Date(),yr=now.getFullYear(),mm=String(now.getMonth()+1).padStart(2,'0'); var curMo=yr+'-'+mm; if(val!==curMo){ fetch('/api/revenue-history?month='+val) .then(function(r){return r.json();}) .then(function(d){ if(d.success&&d.reps){ window._mtdRevReps=d.reps; addRevToCards(); } }); } else { window._mtdRevReps=null; setTimeout(addRevToCards,100); } }; // Run once after data loads setTimeout(addRevToCards,1200);
